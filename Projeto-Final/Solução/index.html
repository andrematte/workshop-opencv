while capture.isOpened():
    # Subtrair o frame atual do frame anterior
    diff = cv2.absdiff(frame1, frame2)

    # Converter para escala de cinza
    gray = cv2.cvtColor(diff, cv2.COLOR_BGR2GRAY)

    blur = cv2.GaussianBlur(gray, (15, 15), 0)

    _, thresh = cv2.threshold(blur, 20, 255, cv2.THRESH_BINARY)

    for contorno in contornos:
        # Desenhar as caixas delimitadoras
        (x, y, w, h) = cv2.boundingRect(contorno)

        # Filtrar contornos com √°rea menor que 3000
        # Ajustar o valor se necess√°rio
        if cv2.contourArea(contorno) < 3000:

        # Desenhar o ret√¢ngulo
        cv2.rectangle(frame1, (x, y), (x + w, y + h), (0, 255, 0), 5)

        # Alerta de Movimento
        cv2.putText(
            frame1,
            "Movimento Detectado!",
            (30, 80),
            cv2.FONT_HERSHEY_PLAIN,
            4,
            (0, 0, 255),
            4,
        )

    # Visualizar frame

    # Pr√≥ximo frame
